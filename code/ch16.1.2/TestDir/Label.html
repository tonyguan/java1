Label 类（标签）

标签（Lebel）的功能是在屏幕上显示单行的字符串，用于提示、说明。

一、谱系：

Java.lang.Object

                 Java.awt.Component

                              Java.awt.Lebel

二、属性

Lebel.CENTER    居中对齐

Lebel.LEFT       向左对齐

Lebel.RIGHT      向右对齐

三、JBuilder6.0环境下的属性窗

name

设置标签在程序中的名字

background

设置标签的背景颜色

font

设置标签的字体属性

foreground

设置标签的字体的颜色

text

设置标签的文字

注：有两种方法来设置背（前）景颜色（background、foreground）。

1．  直接在颜色列表框中选择一种颜色，其中有多种颜色如有红色，白色，黑色等，单击要选的颜色即可。

2．  可在用户颜色设置面板中对颜色进行混合选取，其中有 RGB,HSB两种设置。

（1）在RGB中，有Red、Green、Blue三种基本原色，原色的参数为0~255，可配置自己喜欢的颜色。

（2）在HSB中，用户可对灰暗度进行选择，灰暗度的参数为0~100。

五、JBuilder6.0环境下的示例1（Frame1.java的源程序）：

package ton;      //包名称，一般要有，初学可省去

 

import java.awt.*;         //输入AWT组件，必须有

import java.awt.event.*;    //输入事件包，因用到AWT事件

public class Frame1 extends Frame {        //声明Frame1类，继承Frame类

  private BorderLayout borderLayout1 = new BorderLayout();     //创建布局方式

  private Label label1 = new Label();          //创建标签对象

  public Frame1() {                           //构造方法，启动事件机制

    enableEvents(AWTEvent.WINDOW_EVENT_MASK);

    try {                   

      jbInit();                //调用jbInit()方法

    }

    catch(Exception e) {

      e.printStackTrace();      

    }

  }

  private void jbInit() throws Exception  {      //JBuilder自动生成的初始化方法

    label1.setText("Hello World！");        //设定标签字样

    this.setSize(new Dimension(400, 300));  // 设定Frame1尺寸宽400，高300

    this.setTitle("Hello World");            //设定标题栏字样

    this.add(label1, BorderLayout.CENTER);  //标签摆在窗口中央

  }

  protected void processWindowEvent(WindowEvent e) {  //窗口事件处理方法

    super.processWindowEvent(e);

    if (e.getID() == WindowEvent.WINDOW_CLOSING) { //如果出现关窗口事件

      System.exit(0);                                //系统退出

    }

  }

}

程序：

package demo;

import java.awt.*;

import java.awt.event.*;

import java.applet.*;

 

public class Demo extends Applet {

  private boolean isStandalone = false;

  private Label label1 = new Label();             //创建标签1

  private TextField textField1 = new TextField();    //创建文本框1

  private Checkbox checkbox1 = new Checkbox();   //创建复选框1

  private TextField textField2 = new TextField();    //创建文本框2

  private Label label2 = new Label();             //创建标签2

  private TextArea textArea1 = new TextArea("",3,10,3); /*创建文本区1；设置参数：3行，10列，无滚动条；只能在这里键入*/

    ……

  private void jbInit() throws Exception {

    label1.setText("密码：");

    this.setLayout(null);

    checkbox1.setLabel("设置回显字符");

    checkbox1.setState(true);

    checkbox1.addItemListener(new java.awt.event.ItemListener() {

      public void itemStateChanged(ItemEvent e) {

        checkbox1_itemStateChanged(e);

      }

    });

    textField2.setColumns(1);

    textField2.setText("*");

    textField2.addFocusListener(new java.awt.event.FocusAdapter() {

      public void focusLost(FocusEvent e) {

        textField2_focusLost(e);

      }

    });

 

    label2.setText("说明：");

    textArea1.setEditable(false);

    textArea1.setText("修改设置回显的字符，则由键盘敲入任何内容只显示该回显字符");

    ……

  }

   void checkbox1_itemStateChanged(ItemEvent e) { //复选框状态改变处理程序必须键入

    char ch;                      //声明标记变量

      if(!checkbox1.getState()){     // 如处于未选中状态

        textField2.setEnabled(false);  //文本框2变灰

        ch=0;                    //标记变量置0，强制转换后为空

        textField1.setEchoChar(ch);   //文本框1屏蔽字符为空，即无屏蔽字符

      }

      else{                        //如复选框处于选中状态

        textField1.setEchoChar(textField2.getText().charAt(0)); /*返回文本框2第1个位置字符作为文本框1的屏蔽字符*/

        textField2.setEnabled(true);    //文本框2使能

      }

    }

  void textField2_focusLost(FocusEvent e) {   //文本框2失去焦点处理程序必须键入

    textField1.setEchoChar(textField2.getText().charAt(0)); /*用文本框2第1个位置字符作为文本框1的屏蔽字符*/

  }

}
